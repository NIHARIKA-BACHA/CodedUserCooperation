 60% Function to compute pairwise error probability
 61    function P = pairwise(Ni, Np, N0, Ec1, Ec2, pe, N)
 62    
 63    P=zeros(Ni,Np);
 64    Prob_Nsum1 = zeros(Ni,Np);
 65    for k=1:Ni
 66      for m=1:Np    
 67        PEP = 0;
 68        Prob_N = zeros(Ni,Np);
 69        Prob_Csum = zeros(Ni,Np);
 70        for e=1:Ni
 71            
 72            %Q function for (dE/2 - b) as a reference point
 73            %Q_func= 1/2* erfc((sqrt(2/N0)*((2*(k*Ec1+m*Ec2) - 4*e*m*sqrt(Ec1*Ec2))/(2*sqrt(k*Ec1+m*Ec2))))/sqrt(2));
 74            %Q_func= 1/2* erfc((sqrt(2/N0)*((2*(k*Ec1+m*Ec2) - 4*sqrt(e*Ec1)*sqrt(k*Ec1+m*Ec2))/(2*sqrt(k*Ec1+m*Ec2))))/sqrt(2));
 75            
 76            %Q function for (dE/2) as a reference point
 77            Q_func= 1/2* erfc((sqrt(2/N0)*(sqrt(k*Ec1+m*Ec2)))/sqrt(2));
 78            
 79            %Q function for (dE - b) as a reference point
 80            %Q_func= 1/2* erfc((sqrt(2/N0)*((2*(k*Ec1+m*Ec2) - e*m*sqrt(Ec1*Ec2))/(sqrt(k*Ec1+m*Ec2))))/sqrt(2));
 81            Prob_C = zeros(Np,Np);
 82            sum_mdash = 0;
 83            
 84            for mdash=1:Np
 85            
 86                sum_ii=0;
 87                for ii=0:mdash %max(mdash,m)
 88%                     m
 89%                     ii
 90%                     Np-m
 91%                     mdash-ii
 92                    %if max(mdash,m) >= ii && Np-max(mdash,m) >= mdash-ii && mdash > ii
 93                    if ii <= m && mdash-ii <= Np-m
 94                        sum_ii = sum_ii + (nchoosek1(m, ii) * nchoosek1(Np-m, mdash-ii));
 95                    end
 96                end
 97                %sum_ii
 98                sum_mddash=0;
 99                for mddash=1:Np
100                
101                    sum_mddash = sum_mddash + N(e, mddash);
102                end
103                %sum_mddash
104                sum_i=0;
105                for i=1:mdash %max(mdash,m)
106                
107                    %if max(mdash,m) >= i && Np-max(mdash,m) >= mdash-i && mdash > i
108                    if i <= m && mdash-i <= Np-m
109                        sum_i = sum_i + (Q_func * ((nchoosek1(m, i) * nchoosek1(Np-m, mdash-i)) / sum_ii));
110                        Prob_C(mdash,i) = (nchoosek1(m, i) * nchoosek1(Np-m, mdash-i)) / sum_ii;
111                    else Prob_C(mdash,i) = 0;
112                    end
113                end
114                
115                %Prob_C
116                %sum_i
117                sum_mdash = sum_mdash + (sum_i * (N(e, mdash)/sum_mddash)); %Include the N/sum_N term or not               
118                Prob_N(e,mdash) = N(e, mdash)/sum_mddash;
119            end
120            Prob_Csum(e,:) = sum(Prob_C,1)./Np;
121%             figure(1)
122%             semilogy(1:length(Prob_Csum),Prob_Csum, 'g-*'), hold on, grid on,
123%             figure(2)
124%             semilogy(1:Np,Prob_N, 'r-*'), hold on, grid on,
125            %Prob_N(e)=sum(Prob_N)/mdash;
126            %sum_mdash
127            PEP = PEP + (nchoosek1(Ni, e)* pe^e * (1-pe)^(Ni-e) * sum_mdash);
128        end
129        %PEP
130        P(k,m) = PEP; % Compute PEP for each k and m and store in P.
131        
132        %Prob_Csum1(k,:) = sum(Prob_Csum,1)./Ni;
133        Prob_Nsum1(k,:) = sum(Prob_N,1)./Ni;
134%         figure(1)
135%         semilogy(1:length(Prob_Csum1),Prob_Csum1, 'g-*'), hold on, grid on,
136%         figure(2)
137%         semilogy(1:Np,Prob_Nsum1, 'r-*'), hold on, grid on,
138      end
139    end
140    %P
141    Prob_Csum2 = sum(Prob_Csum,1)./Ni;
142    Prob_Nsum2 = sum(Prob_Nsum1,1)./Ni;
143    %figure(1)
144    %semilogy(1:length(Prob_Csum2),Prob_Csum2, 'g-*'), hold on, grid on,
145    %figure(2)
146    %semilogy(1:length(Prob_Nsum2),Prob_Nsum2, 'r-*'), hold on, grid on,
147    end
