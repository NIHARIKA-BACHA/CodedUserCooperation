149% Function to compute the distribution of codewords
150    function N = codeweight(Ni, Np, d)
151
152        N=zeros(Ni,Np);
153        Pjk=zeros(1,Ni);  
154
155        for k=1:Ni
156            p=0;
157            for j = max(0,k+d-Ni): min(d,k)
158            if mod(j,2) ~= 0
159                p = p + ((nchoosek1(d,j) * nchoosek1((Ni-d),(k-j)))/nchoosek1(Ni,k));
160            end
161            end
162            Pjk(k)=p; %Prob. of j=1, given k 1's in the information vector
163        end
164
165        for k=1:Ni
166            for m=1:Np
167             N(k,m) = nchoosek1(Ni,k)*nchoosek1(Np,m)*Pjk(k)^m*(1-Pjk(k))^(Np-m); %Total no. of codewords with input (information bits) weight k, and output (parity bits) weight m 
168            end
169        end
170    end
