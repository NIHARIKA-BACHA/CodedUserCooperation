1function Pe = cuc_errorbound1() %#eml
  2
  3%clear all;
  4%close all;
  5
  6Ni = 50;
  7Np = 50;
  8d = 7;
  9w = 1:Ni+Np;
 10SNR = 0:10;
 11pbar=0;
 12pe = 0; %BSC crossover probability
 13%error_in = 0:Ni; % Range of errors in input Ni bits
 14%error_out = 0:Np; % Range of errors in output Np bits
 15Pe=zeros(1,length(SNR));
 16Pe1=0;
 17
 18%Codeweight distribution
 19N = codeweight(Ni, Np, d);
 20%N
 21for x=1:length(SNR)
 22% EsN0 calculation
 23Ec1 = 1;
 24Ec2 = 1;
 25EbN0 = 10^(SNR(x)/10);
 26EsN0 = EbN0;
 27N0 = (1/EbN0)*Ec1; % Noise variance
 28
 29% Compute the Pe for first phase BSC channel error
 30pbar=1/2*(1-sqrt((Ec1*1/N0)/(1+(Ec1*1/N0))));
 31%pbar=1/2*(1-sqrt(SNR(x)/(1+SNR(x))));
 32%pbar=1/2*erfc(sqrt(Ec1/N0));
 33%pbar=0.01;
 34pe=(1-(1-2*pbar)^d)/2;
 35
 36%Pairwise error probability
 37P = pairwise(Ni, Np, N0, Ec1, Ec2, pe, N);
 38
 39
 40Pe1=0;
 41for k=1:Ni
 42    for m=1:Np
 43      %Pe1 
 44      Pe1 = Pe1 + N(k,m)*P(k,m);%* (1/2* erfc(2*sqrt((k*Ec1+m*Ec2)/(2*N0))));
 45    end
 46   
 47end
 48Pe(x) = Pe1
 49%fprintf('\nSimulation done for SNR: %d\n',SNR(x));
 50end
 51%figure(3)
